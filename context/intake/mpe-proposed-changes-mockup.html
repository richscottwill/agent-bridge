// mpe_narrative.js — add renderSinceLastWeek() before renderNarrative() function renderSinceLastWeek(wow, regimes, out) { const brandTag = tagFor(wow.brand_regs_pct); const nbTag = tagFor(wow.nb_regs_pct); const cpaTag = tagFor(wow.cpa_pct, /*invert*/ true); // lower CPA = good const activeLifts = regimes.filter(r => !r.absorbed_into_baseline && !r.low_confidence); const liftsNote = activeLifts.length > 0 ? `The ${activeLifts[0].name || 'active'} lift is holding — no decay detected.` : 'No active campaign lifts this week.'; const driver = identifyWeekDriver(wow); // "Non-Brand spend pacing ahead of plan" etc. return `<strong>Since last week:</strong> ` + `Brand regs ${brandTag}, NB regs ${nbTag}, blended CPA ${cpaTag}.<br>` + `${liftsNote} Primary driver this week is ${driver}.`; } function tagFor(pct, invert = false) { const cls = (pct >= 0) === !invert ? 'up' : 'down'; const sign = pct >= 0 ? '+' : ''; return `<span class="delta-tag ${cls}">${sign}${pct.toFixed(1)}%</span>`; }
Acceptance: Directly above the Narrative heading, one-paragraph summary auto-generated from WoW data. Green/red pill tags. Identifies the primary driver of the week's change.

// mpe_narrative.js — renderHeroContext() or an inline helper function ciQualityLabel(loPct, hiPct) { // loPct/hiPct are absolute widths relative to central (e.g., 0.37 = 37% below central) const width = loPct + hiPct; if (width < 0.20) return { cls: 'conf-high', label: 'tight · high confidence' }; if (width < 0.40) return { cls: 'conf-medium', label: 'moderate' }; return { cls: 'conf-low', label: 'wide · low confidence' }; } function planInRange(op2Value, ciLo, ciHi) { if (op2Value < ciLo) return { flag: 'below', cls: 'plan-in-range-no', msg: 'OP2 plan falls below 90% CI — projection meaningfully above plan.' }; if (op2Value > ciHi) return { flag: 'above', cls: 'plan-in-range-no', msg: 'OP2 plan falls above 90% CI — projection meaningfully below plan.' }; // Within range — flag edge cases const position = (op2Value - ciLo) / (ciHi - ciLo); if (position < 0.2) return { flag: 'low-edge', cls: 'plan-in-range-yes', msg: `OP2 plan ($${fmt(op2Value)}) at low edge of range. Within 90% CI but projection signals overspend risk.` }; if (position > 0.8) return { flag: 'high-edge', cls: 'plan-in-range-yes', msg: `OP2 plan ($${fmt(op2Value)}) at high edge of range. Within 90% CI but projection signals underdelivery risk.` }; return { flag: 'center', cls: 'plan-in-range-yes', msg: 'Plan comfortably within 90% CI.' }; } function contingencyNudge(central, ciHi) { const contingency = ciHi - central; return `If you commit to the point estimate, prepare a ±$${fmtShort(contingency)} contingency band.`; }
Acceptance: Hero context line has (1) the range, (2) a confidence-label pill (tight/moderate/wide), (3) a plan-in-range sentence naming the OP2 value and its position, (4) a contingency nudge. All four surface without expanding a disclosure.
